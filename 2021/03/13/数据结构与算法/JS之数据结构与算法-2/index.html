集合

要点
  • 一种无序且唯一的数据结构
  • ES6中有集合，名为Set
  • 集合的常用操作：去重、判断元素是否在集合中、求交集……
相关操作
//去重
const arr = [1, 1, 2, 2]
const arr1 = [...new Set(arr)];
console.log(arr1);//[1,2]

//判断元素是否在集合中
const set = new Set(arr);
const has = set.has(3);
console.log(has);//false

//求交集
const set2 = new Set([2, 3]);
const set3 = new Set([...set].filter(item => set2.has(item)));
console.log([...set3]);//[2]

//求差集,也就是求set2里面没有而set里面有的值
const set4 = new Set([...set].filter(item => !set2.has(item)))
Javascript与集合

  • 集合的增删查

    let mySet = new Set();

//add()方法可以往集合内添加元素，原始值是唯一得，索引值得的地址也是唯一的
mySet.add(1);
mySet.add('str');
mySet.add({ a: 1, b: 2 });

//has()方法可以查看集合中是否存在某个元素
const has = mySet.has('str');//true

//delete()方法可以删除集合中的某个元素
mySet.delete('str')

  • 集合的迭代

    //利用for of，和集合的value()或者key()方法，下面的value()可以换为key(),两者返回的值都是一样的
for(let item of mySet.values()){
   console.log(item);
}

  • Set与Array的互转

    //Set→Array
const myArr = [...mySet];
const myArr2 = Array.from(mySet);

//Array→Set
const mySet2 = new Set([1,2,3,4])
相关算法题

  • 两个数组的交集

    /**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var intersection = function(nums1, nums2) {
    // const set1 = new Set(nums1);
    const set = new Set(nums2);
    return [...new Set(nums1.filter(item => set.has(item)))]
};

//解题思路：利用数组的filter方法和Set的has方法可直接得到结果
//也可以直接用数组的includes方法：
return [...new Set(nums1.filter(item => nums2.includes(item)))]

字典

要点：
  • 与集合类似，字典也是一种存储唯一值的数据结构，但是它是以键值对的形式来存储
  • ES6中有字典的数据结构，名为Map
  • 字典的常用操作：键值对的增删改查
字典的常用操作
//创建一个字典
const m = new Map();

//增
m.set('a','b');

//查
m.get('a');//'b'
m.has('a');//true

//删
m.delete('a');
//清除字典中所有的键值对
m.clear();

//改
m.set('a','c');

  • 先序遍历

    //思路： （根、左、右）
//访问根节点
//对根节点的左子树进行先序遍历
//对根节点的右子树进行先序遍历

//递归
const preorder = (root) => {
    if (!root) { return; }
    console.log(root.val);
    preorder(root.left);
    preorder(root.right);
}

//非递归
const preorder = (root) => {
    if (!root) { return; }
    const stack = [root];
    while (stack.length) {
        const n = stack.pop();
        console.log(n.val);
       //由于栈是后进先出，所以左节点后进
        if (n.right) stack.push(n.right);
        if (n.left) stack.push(n.left);
    }
}

  • 中序遍历

    //思路 （左、根、右）
//对根节点的左子树进行中序遍历
//访问根节点
//对根节点的右子树进行中序遍历

//递归
const inorder = (root) => {
    if (!root) { return; }
    inorder(root.left);
    console.log(root.val);
    inorder(root.right);
}

//非递归
const inorder = (root) => {
    if (!root) { return; }
    const stack = [];
    let p = root;
    while (stack.length || p) {
        while (p) {
            stack.push(p);
            p = p.left;
        }
        const n = stack.pop();
        console.log(n.val);
        p = n.right;
    }
}

  • 后序遍历

    //思路 （左、右、根）
//对根节点的左子树进行后续遍历
//对根节点的右子树进行后续遍历
//访问根节点

//递归
const postorder = (root) => {
    if (!root) { return; }
    postorder(root.left);
    postorder(root.right);
    console.log(root.val);
}

//非递归
const postorder = (root) => {
    if (!root) { return; }
    const outputStack = [];
    const stack = [root];
    while (stack.length) {
        const n = stack.pop();
        outputStack.push(n);
        if (n.left) stack.push(n.left);
        if (n.right) stack.push(n.right);
    }
    while (outputStack.length) {
        const n = outputStack.pop();
        console.log(n.val)
    }
}

postorder(bt);

  • 二叉树的最小深度

    var minDepth = function(root) {
    if(!root) return 0;
    const q = [[root,1]];
    while(q.length){
    const [n,height] = q.shift();
    if(!n.left && !n.right){
        return height;
    }
    if(n.left) q.push([n.left,height+1]);
    if(n.right) q.push([n.right,height+1]);
    } 
};

//利用广度优先算法，若遍历到叶子节点（即没左右子节点的节点），则直接返回该叶子节点所在层数即可。

  • 二叉树的层序遍历

    var levelOrder = function(root) {
    if(!root) {return [];}
    const q = [[root,0]];
    const res = [];
    while(q.length){
        const [n, height] = q.shift();
        if(!res[height]){
            res.push([n.val])
        }else{
            res[height].push(n.val);
        }
        if(n.left) q.push([n.left, height + 1]);
        if(n.right) q.push([n.right, height + 1]);
    }
    return res;
};

//使用广度优先遍历并记录每个元素所在层级，然后将相同层级的元素push进一个数组

  • 二叉树的中序遍历

    //递归版，较简单
var inorderTraversal = function(root) {
    if(!root) return [];
    let res = [];
    const inorder = (n) =>{
        if(!n) return;
        inorder(n.left);
        res.push(n.val);
        inorder(n.right);
    }
    inorder(root);
    return res;
};

//迭代版，稍复杂，建议断点调试多看看
var inorderTraversal = function(root) {
    const res = [];
    const stack = [];
    let p = root;
    while(stack.length || p){
        while(p){
            stack.push(p);
            p = p.left;
        }
        const n = stack.pop();
        res.push(n.val);
        p = n.right;
    }
    return res;
};

  • 二叉树的路径总和

    var hasPathSum = function(root, targetSum) {
    if(!root) return false;
    let res = false;
    const dfs = (n, s) =>{
        if(!n.left && !n.right && s === targetSum){
            res = true;
        }
        if(n.left) dfs(n.left, s + n.left.val);
        if(n.right) dfs(n.right, s + n.right.val);
    }
    dfs(root, root.val);
    return res;    
};

//思路：核心是深度优先遍历，在遍历每个节点时，记录下当前节点的路径总和，当遍历到叶子节点时，判断此时路径总和是否为targetSum
